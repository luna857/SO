📧 7. SENDGRID EN PRODUCCIÓN

📌 instalar:

📌 configuración:

const sgMail = require("@sendgrid/mail");

sgMail.setApiKey(process.env.SENDGRID_API_KEY);

function sendEmail(email, link) {
  return sgMail.send({
    to: email,
    from: "tuemail@gmail.com",
    subject: "Verificación",
    text: `Verifica tu cuenta: ${link}`
  });
}

👤 8. findOneOrCreateByGoogle

📌 models/User.js

UserSchema.statics.findOneOrCreateByGoogle = async function(profile) {
  let user = await this.findOne({ email: profile.emails[0].value });

  if (user) return user;

  user = new this({
    email: profile.emails[0].value,
    verificado: true
  });

  await user.save();
  return user;
};
🔐 9. VALIDACIÓN TOKEN FACEBOOK

📌 middleware:

function verifyFacebookToken(req, res, next) {
  const token = req.headers["authorization"];

  if (!token) return res.status(403).send("No token");

  // aquí iría validación con Facebook Graph API
  next();
}
