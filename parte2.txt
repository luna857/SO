  nombre: String,
  color: String,
  lat: Number,
  lng: Number
});

module.exports = mongoose.model("Bicicleta", BicicletaSchema);

4. API CRUD (routes/bicicletas.js)

const express = require("express");
const router = express.Router();
const Bicicleta = require("../models/Bicicleta");

// GET
router.get("/bicicletas", async (req, res) => {
  const bikes = await Bicicleta.find();
  res.json(bikes);
});

// POST
router.post("/bicicletas", async (req, res) => {
  const bike = new Bicicleta(req.body);
  await bike.save();
  res.json(bike);
});

// DELETE
router.delete("/bicicletas/:id", async (req, res) => {
  await Bicicleta.findByIdAndDelete(req.params.id);
  res.json({ msg: "eliminada" });
});

module.exports = router;



. TEST API (spec/bicicleta_api_test.spec.js)
const request = require("request");

const base_url = "http://localhost:3000/api/bicicletas";

describe("API Bicicletas", () => {

  it("GET devuelve bicicletas", (done) => {
    request.get(base_url, (err, res) => {
      expect(res.statusCode).toBe(200);
      done();
    });
  });

  it("POST crea bicicleta", (done) => {
    request.post({
      url: base_url,
      json: { nombre: "Bike Test", lat: 14, lng: -87 }
    }, (err, res) => {
      expect(res.statusCode).toBe(200);
      done();
    });
  });

});
